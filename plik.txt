import java.util.Arrays;
import java.util.Scanner;

public class AnagramyJava {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Podaj pierwsza liczbe: ");
        int liczba1 = scanner.nextInt();
        System.out.println("Podaj druga liczbe: ");
        int liczba2 = scanner.nextInt();
        scanner.close();
        
        // Konwersja liczb na tablice znakow
        char[] liczba1Chars = Integer.toString(liczba1).toCharArray();
        char[] liczba2Chars = Integer.toString(liczba2).toCharArray();
        
        // Sortowanie znakow w tablicach
        Arrays.sort(liczba1Chars);
        Arrays.sort(liczba2Chars);
        
        // Por√≥wnanie posortowanych tablic znakow
        if (Arrays.equals(liczba1Chars, liczba2Chars)) {
            System.out.println("Podane liczby sa anagramami.");
        } else {
            System.out.println("Podane liczby nie sa anagramami.");
        }
    }
}
